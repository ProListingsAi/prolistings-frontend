// or the member whose role contains 'Senior' and was added first (id=1).

// You can also explicitly set m.isAdmin = true on a member.

// Admin PIN = their own login PIN.


function getAdminMember() {

  // First look for explicit admin flag

  for (var i = 0; i < TEAM.length; i++) {

    if (TEAM[i].isAdmin) return TEAM[i];

  }

  // Fall back to Team Manager role

  for (var i = 0; i < TEAM.length; i++) {

    if ((TEAM[i].role || '').indexOf('Manager') > -1) return TEAM[i];

  }

  // Fall back to lowest ID (first member added)

  if (TEAM.length > 0) {

    var sorted = TEAM.slice().sort(function(a,b){ return a.id - b.id; });

    return sorted[0];

  }

  return null;

}

// ── Save member ──────────────────────────────────────────────

function saveMember() {

  var first = document.getElementById('m-first').value.trim();

  var last  = document.getElementById('m-last').value.trim();


  document.getElementById('m-first').style.borderColor = first ? '' : 'var(--red)';

  document.getElementById('m-last').style.borderColor  = last  ? '' : 'var(--red)';

  if (!first || !last) return;


  var data = {

    first: first,

    last:  last,

    role:  document.getElementById('m-role').value,

    phone: document.getElementById('m-phone').value.trim(),

    email: document.getElementById('m-email').value.trim(),

    pin:   document.getElementById('m-pin').value.trim(),

    bg:    selBg,

    fg:    selFg

  };


  var eid = editingId || parseInt(document.getElementById('edit-id').value) || 0;


  if (eid) {

    // Update existing — find by ID and replace in place

    var found = false;

    for (var i = 0; i < TEAM.length; i++) {

      if (TEAM[i].id === eid) {

        TEAM[i] = Object.assign({}, TEAM[i], data);

        found = true;

        break;

      }

    }

    if (!found) {

      // Fallback: add as new if somehow not found

      data.id = getNextId(); data.calls = 0; data.leads = 0; data.closed = 0;

      TEAM.push(data);

    }

  } else {

    // New member — get a fresh unique ID

    data.id = getNextId(); data.calls = 0; data.leads = 0; data.closed = 0;

    TEAM.push(data);

  }


  editingId = null;

  saveTeam();

  renderTeam();

  closeM('member-modal');

}

function getCredentials() {

function attemptLogin() {

  var email = (document.getElementById('login-email').value || '').trim().toLowerCase();

  var pin   = (document.getElementById('login-pin').value || '').trim();

  var errEl = document.getElementById('login-error');


  if (!email || !pin) {

    errEl.textContent = 'Please enter your email and PIN.';

    errEl.classList.add('show');

    return;

  }


  // Find matching team member

  var match = null;

  for (var i = 0; i < TEAM.length; i++) {

    var m = TEAM[i];

    if ((m.email || '').toLowerCase() === email && (m.pin || '') === pin) {

      match = m;

      break;

    }

  }


  if (!match) {

    errEl.textContent = 'Incorrect email or PIN. Please try again.';

    errEl.classList.add('show');

    document.getElementById('login-pin').value = '';

    return;

  }


  // Successful login

  errEl.classList.remove('show');

  currentUser = match;


  // Save session to localStorage so they stay logged in on refresh

  _ls.setItem('pl_session', JSON.stringify({id: match.id, ts: Date.now()}));


  // Mark member as online

  markOnline(match.id, true);


  showApp();

}


function showApp() {

  var ls = document.getElementById('login-screen');

  var aw = document.getElementById('app-wrap');

  if (ls) { ls.classList.remove('visible'); ls.classList.add('hidden'); ls.style.display = 'none'; }

  renderCurrentUser();

}


function logout() {

  currentUser = null;

  var ls = document.getElementById('login-screen');

  var aw = document.getElementById('app-wrap');

  var em = document.getElementById('login-email'); if (em) em.value = '';

  var pm = document.getElementById('login-pin');   if (pm) pm.value = '';

}


function restoreSession() {

  var saved = _ls.getItem('pl_session');

  if (!saved) return false;

  try {

    var sess = JSON.parse(saved);

    // Session expires after 12 hours

    if (Date.now() - sess.ts > 12 * 60 * 60 * 1000) {

      _ls.removeItem('pl_session');

      return false;

    }

    for (var i = 0; i < TEAM.length; i++) {

      if (TEAM[i].id === sess.id) {

        currentUser = TEAM[i];

        markOnline(TEAM[i].id, true);

        return true;

      }

    }

  } catch(e) {}

  return false;

}

// Online/offline tracking — stored in localStorage with timestamps

function markOnline(memberId, isOnline) {

  var key = 'pl_online_' + memberId;

  if (isOnline) {

    _ls.setItem(key, Date.now().toString());

  } else {

    _ls.removeItem(key);

  }

  renderSidebar();

}


function isOnline(memberId) {

  var key = 'pl_online_' + memberId;

  var ts  = _ls.getItem(key);

  if (!ts) return false;

  // Consider online if last seen within 15 minutes

  return (Date.now() - parseInt(ts)) < 15 * 60 * 1000;

}
